REM Windows Host Discovery (Local Subnet)
DELAY 1000
GUI r
DELAY 500
STRING powershell -NoProfile -WindowStyle Hidden -Command "$ip = (Get-NetIPAddress -AddressFamily IPv4 | Where-Object { $_.InterfaceAlias -notmatch 'Loopback' }).IPAddress[0]; $subnet = $ip.Substring(0, $ip.LastIndexOf('.')); 1..254 | ForEach-Object { if (Test-Connection -ComputerName \"$subnet.$_\" -Count 1 -Quiet) { \"$subnet.$_\" } } | Out-File -FilePath %TEMP%\hosts.txt"
ENTER
